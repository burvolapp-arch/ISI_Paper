% --- Convenience Shortcuts ---------------------------------------------------
\newcommand{\todo}[1]{\textcolor{red}{\textbf{TODO:}~#1}}  % Inline TODO marker
\newcommand{\fixme}[1]{\textcolor{orange}{\textbf{FIXME:}~#1}}
\newcommand{\note}[1]{\textcolor{blue}{\textbf{NOTE:}~#1}}

% --- Figure Insertion Helper -------------------------------------------------
% Usage: \insfig{filename}{width}{caption}{label}
%   filename: relative to figures/ (no prefix needed — \graphicspath handles it)
\newcommand{\insfig}[4]{%
  \begin{figure}[htbp]
    \centering
    \includegraphics[width=#2\textwidth]{#1}%
    \caption{#3}%
    \label{fig:#4}%
  \end{figure}%
}

% --- Table from External File Helper -----------------------------------------
% Usage: \instab{caption}{filename}{label}
%   filename: file in tables/ directory (e.g. my-table.tex)
\newcommand{\instab}[3]{%
  \begin{table}[htbp]
    \centering
    \caption{#1}%
    \label{tab:#3}%
    \input{tables/#2}%
  \end{table}%
}
